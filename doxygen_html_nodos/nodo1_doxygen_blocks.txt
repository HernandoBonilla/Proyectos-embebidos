/**
 * @file main.c
 * @brief Programa principal del Nodo 1.
 * @details Inicializa periféricos locales, activa ESP-NOW como receptor,

 */

 * @details Configura NVS, netif, event loop, WiFi en modo STA y canal fijo.
 */

/**

/**
 * @brief Tarea de lectura del sensor de pulso.
 */
static void task_pulse(void *pv);

/**
 * @brief Tarea de lectura del IMU local.
 */
static void task_imu(void *pv);

/**
 * @brief Tarea de lectura del RTC.
 */
static void task_rtc(void *pv);

/**
 * @brief Tarea de envío por serial al PC.
 * @details Genera una trama consolidada con datos locales y remotos.
 */
static void task_serial(void *pv);

 * @brief Tarea de actualización de la pantalla LCD.
 */
static void task_display(void *pv);
